84  * Construct basis for pseudo-spectral scft.
85  */
86  template <int D>
87  void Basis<D>::makeBasis(const Mesh<D>& mesh,
88  const UnitCell<D>& unitCell,
89  const SpaceGroup<D>& group)
90  {
91  // Save pointers to mesh and unit cell
92  meshPtr_ = &mesh;
93  unitCellPtr_ = &unitCell;
94 
95  // Allocate arrays
96  nWave_ = mesh.size();
97  waves_.allocate(nWave_);
98  waveIds_.allocate(nWave_);
99 
100  // Make sorted array of waves
101  makeWaves();
102 
103  // Identify stars of waves that are related by symmetry
104  makeStars(group);
105 
106  // Apply validity test suite
107  bool valid = isValid();
108  if (!valid) {
109  UTIL_THROW("Basis failed validity check");
110  }
111 
112  }
113 
114  /*
115  * Construct ordered list of waves.
116  *
117  * On exit:
118  * - Array waves_ contains list of waves ordered by sqNorm.
119  * - Each wave has indicesDft, indicesBz and sqNorm set.
120  * - Array stars_ is still empty.
121  */
122  template <int D>
123  void Basis<D>::makeWaves()
124  {
125  IntVec<D> meshDimensions = mesh().dimensions();
126  std::vector< TWave<D> > twaves;
127  twaves.reserve(nWave_);
128 
129  // Loop over dft mesh to generate all waves, add to twaves
130  TWave<D> w;
131  IntVec<D> v;
132  MeshIterator<D> itr(mesh().dimensions());
133  for (itr.begin(); !itr.atEnd(); ++itr) {
134  w.indicesDft = itr.position();
135  v = shiftToMinimum(w.indicesDft, meshDimensions, *unitCellPtr_);
136  w.indicesBz = v;
137  w.sqNorm = unitCell().ksq(v);
138  twaves.push_back(w);
139  }
140 
141  // Sort temporary array twaves
142  TWaveNormComp<D> comp;
143  std::sort(twaves.begin(), twaves.end(), comp);
144 
145  // Copy temporary array twaves into member variable waves_
146  for (int i = 0; i < nWave_; ++i) {
147  waves_[i].sqNorm = twaves[i].sqNorm;
148  waves_[i].indicesDft = twaves[i].indicesDft;
149  waves_[i].indicesBz = twaves[i].indicesBz;
150  }
151 
152  }

154  template <int D>
155  void Basis<D>::makeStars(const SpaceGroup<D>& group)
156  {
157  /*
158  * Local containers that hold TWave<D> objects:
159  * list - set of waves of equal norm, compared by indicesDft
160  * star - set of symmetry-related waves, compared by indicesDft
161  * tempStar - temporary star, sorted by descending indicesBz
162  * tempList - temporary list, ordered by star
163  */
164  std::set< TWave<D>, TWaveDftComp<D> > list;
165  std::set< TWave<D>, TWaveDftComp<D> > star;
166  std::vector< TWave<D> > tempStar;
167  GArray< TWave<D> > tempList;
168 
169  typename std::set< TWave<D>, TWaveDftComp<D> >::iterator rootItr;
170  typename std::set< TWave<D>, TWaveDftComp<D> >::iterator setItr;
171  TWave<D> wave;
172  Basis<D>::Star newStar;
173 
174  std::complex<double> coeff;
175  double Gsq;
176  double Gsq_max;
177  double phase_diff;
178  const double twoPi = 2.0*Constants::Pi;
179  const double epsilon = 1.0E-3;
180  IntVec<D> meshDimensions = mesh().dimensions();
181  IntVec<D> rootVecBz;
182  IntVec<D> vec;
183  IntVec<D> nVec;
184  int listId = 0; // id for this list
185  int listBegin = 0; // id of first wave in this list
186  int listEnd = 0; // (id of last wave in this list) + 1
187  int starId = 0; // id for this star
188  int starBegin = 0; // id of first wave in this star
189  int i, j, k;
190  bool cancel;
191 
192  // Loop over all waves
193  nBasis_ = 0;
194  nBasisWave_ = 0;
195  Gsq_max = waves_[0].sqNorm;
196  for (i = 1; i <= nWave_; ++i) {
197 
198  // Determine if this wave begins a new list
199  bool newList = false;
200  if (i == nWave_) {
201  listEnd = i;
202  newList = true;
203  } else {
204  Gsq = waves_[i].sqNorm;
205  if (Gsq > Gsq_max + epsilon) {
206  Gsq_max = Gsq;
207  listEnd = i;
208  newList = true;
209  }
210  }
211 
212  // Process completed list of wavectors of equal norm
213  if (newList && listEnd > 0) {
214 
215  // Copy waves of equal norm into std::set "list"
216  list.clear();
217  tempList.clear();
218  for (j = listBegin; j < listEnd; ++j) {
219  wave.indicesDft = waves_[j].indicesDft;
220  wave.indicesBz = waves_[j].indicesBz;
221  wave.sqNorm = waves_[j].sqNorm;
222  if (j > listBegin) {
223  UTIL_CHECK( abs(wave.sqNorm-waves_[j].sqNorm)
224  < 2.0*epsilon );
225  }
226  list.insert(wave);
227  }
228 
229  // On entry to each iteration of the loop over stars,
230  // rootItr and nextInvert are known. The iterator rootItr
231  // points to the wave in the remaining list that will be
232  // used as the root of the next star. The flag nextInvert
233  // is equal to -1 iff the previous star was the first of
234  // a pair that are open under inversion, and is equal to +1
235  // otherwise.
236 
237  // Initial values for first star in this list
238  rootItr = list.begin();
239  int nextInvert = 1;
240 
241  // Loop over stars with a list of waves of equal norm,
242  // removing each star from the list as it is identified.
243  // The root of the next star must have been chosen on
244  // entry to each iteration of this loop.
245 
246  while (list.size() > 0) {
247 
248  rootVecBz = rootItr->indicesBz;
249  Gsq = rootItr->sqNorm;
250  cancel = false;
251  star.clear();
252 
253  // Construct a star from root vector, by applying every
254  // symmetry operation in the group to the root wavevector.
255  for (j = 0; j < group.size(); ++j) {
256 
257  // Apply symmetry (i.e., multiply by rotation matrix)
258  // vec = rotated wavevector.
259  vec = rootVecBz*group[j];
260 
261  // Check that rotated vector has same norm as root.
262  UTIL_CHECK(abs(Gsq - unitCell().ksq(vec)) < epsilon);
263 
264  // Initialize TWave object associated with rotated wave
265  wave.sqNorm = Gsq;
266  wave.indicesBz = shiftToMinimum(vec, meshDimensions,
267  *unitCellPtr_);
268  wave.indicesDft = vec;
269  mesh().shift(wave.indicesDft);
270 
271  // Compute phase for coeff. of wave in basis function.
272  // Convention -pi < phase <= pi.
273  wave.phase = 0.0;
274  for (k = 0; k < D; ++k) {
275  wave.phase += rootVecBz[k]*(group[j].t(k));
276  }
277  while (wave.phase > 0.5) {
278  wave.phase -= 1.0;
279  }
280  while (wave.phase <= -0.5) {
281  wave.phase += 1.0;
282  }
283  wave.phase *= twoPi;
284 
285  // Check for cancellation of star: The star is
286  // cancelled if application of any symmetry operation
287  // in the group to the root vector yields a rotated
288  // vector equivalent to the root vector but with a
289  // nonzero phase, creating a contradiction.
290 
291  if (wave.indicesDft == rootItr->indicesDft) {
292  if (abs(wave.phase) > 1.0E-10) {
293  cancel = true;
294  }
295  }
296 
297  // Search for an equivalent wave already in the star.
298  // Note: Equivalent waves have equal DFT indices.
299  setItr = star.find(wave);
300 
301  if (setItr == star.end()) {
302 
303  // If no equivalent wave is found in the star,
304  // then add this wave to the star
305  star.insert(wave);
306 
307  } else {
308 
309  // If an equivalent wave is found, check if the
310  // phases are equivalent. If not, the star is
311  // cancelled.
312 
313  phase_diff = setItr->phase - wave.phase;
314  while (phase_diff > 0.5) {
315  phase_diff -= 1.0;
316  }
317  while (phase_diff <= -0.5) {
318  phase_diff += 1.0;
319  }
320  if (abs(phase_diff) > 1.0E-10) {
321  cancel = true;
322  }
323 
324  }
325 
326  }
327 
328  // Copy all waves from set star to std::vector tempStar
329  tempStar.clear();
330  setItr = star.begin();
331  for ( ; setItr != star.end(); ++setItr) {
332  tempStar.push_back(*setItr);
333  }
334 
335  // Sort tempStar, in descending order by indicesBz.
336  TWaveBzComp<D> waveBzComp;
337  std::sort(tempStar.begin(), tempStar.end(), waveBzComp);
338 
339  // Append contents of tempStar to tempList, erase from list
340  int tempStarSize = tempStar.size();
341  for (j = 0; j < tempStarSize; ++j) {
342  list.erase(tempStar[j]);
343  tempList.append(tempStar[j]);
344  }
345  UTIL_CHECK((int)(tempList.size()+list.size())==listEnd-listBegin);
346 
347  // If this star is not cancelled, increment the number
348  // of basis functions (nBasis_) and waves (nBasisWave_)
349  if (!cancel) {
350  ++nBasis_;
351  nBasisWave_ += star.size();
352  }
353 
354  // Initialize a Star object
355  newStar.eigen = Gsq;
356  newStar.beginId = starBegin;
357  newStar.endId = newStar.beginId + star.size();
358  newStar.size = star.size();
359  newStar.cancel = cancel;
360  // Note: newStar.starInvert is not yet known
361 
362  // Determine invertFlag, rootItr and nextInvert
363  if (nextInvert == -1) {
364 
365  // If this star is 2nd of a pair related by inversion,
366  // set root of next star to 1st wave of remaining list.
367 
368  newStar.invertFlag = -1;
369  rootItr = list.begin();
370  nextInvert = 1;
371 
372  } else {
373 
374  // If this star is not the 2nd of a pair of partners,
375  // then determine if it is closed under inversion.
376 
377  // Compute negation nVec of root vector in FBZ
378  nVec.negate(rootVecBz);
379 
380  // Shift negation nVec to a DFT mesh
381  (*meshPtr_).shift(nVec);
382 
383  // Search for negation of root vector within this star
384  bool negationFound = false;
385  setItr = star.begin();
386  for ( ; setItr != star.end(); ++setItr) {
387  if (nVec == setItr->indicesDft) {
388  negationFound = true;
389  break;
390  }
391  }
392 
393  if (negationFound) {
394 
395  // If this star is closed under inversion, the root
396  // of next star is the 1st vector of remaining list.
397 
398  newStar.invertFlag = 0;
399  rootItr = list.begin();
400  nextInvert = 1;
401 
402  } else {
403 
404  newStar.invertFlag = 1;
405  nextInvert = -1;
406 
407  // If star is not closed, find negation of root in
408  // the remaining list, and use this negation as the
409  // root of the next star.
410 
411  setItr = list.begin();
412  for ( ; setItr != list.end(); ++setItr) {
413  if (nVec == setItr->indicesDft) {
414  negationFound = true;
415  rootItr = setItr;
416  break;
417  }
418  }
419  // If negationFound, then rootItr->indicesDft = nVec
420 
421  // Failure to find the negation here is an error:
422  // It should be either in this star or remaining list
423 
424  if (!negationFound) {
425  std::cout << "Negation not found for: " << "\n";
426  std::cout << " vec (ft):"
427  << rootItr->indicesDft <<"\n";
428  std::cout << " vec (bz):"
429  << rootItr->indicesBz <<"\n";
430  std::cout << "-vec (dft):" << nVec << "\n";
431  UTIL_CHECK(negationFound);
432  }
433 
434  }
435 
436  }
437 
438  stars_.append(newStar);
439  ++starId;
440  starBegin = newStar.endId;
441 
442  }
443  // End loop over stars within a list.
444  // Processing of list is now complete.
445 
446  UTIL_CHECK(list.size() == 0);
447  UTIL_CHECK(tempList.size() == listEnd - listBegin);
448 
449  // Copy tempList into corresponding section of waves_,
450  // overwriting the section of waves_ used to create the list.
451  // Compute a complex coefficient of unit norm for each wave.
452  for (j = 0; j < tempList.size(); ++j) {
453  k = j + listBegin;
454  waves_[k].indicesDft = tempList[j].indicesDft;
455  waves_[k].indicesBz = tempList[j].indicesBz;
456  waves_[k].sqNorm = tempList[j].sqNorm;
457  coeff = std::complex<double>(0.0, tempList[j].phase);
458  coeff = exp(coeff);
459  if (abs(imag(coeff)) < 1.0E-10) {
460  coeff = std::complex<double>(real(coeff), 0.0);
461  }
462  if (abs(real(coeff)) < 1.0E-10) {
463  coeff = std::complex<double>(0.0, imag(coeff));
464  }
465  waves_[k].coeff = coeff;
466  }
467 
468  // At this point, waves_[k].coeff has unit absolute magnitude,
469  // and correct relative phases for waves within a star, but
470  // the coeff may not be unity for the first or last wave in
471  // the star.
472 
473  ++listId;
474  listBegin = listEnd;
475  }
476  // Finished processing a list of waves of equal norm
477 
478  } // End loop over all waves
479  nStar_ = stars_.size();
480  // Complete initial processing of all lists and stars
481 
482  std::complex<double> rootCoeff; // Coefficient of root wave
483  std::complex<double> partCoeff; // Coefficient of partner of root
484  std::complex<double> d;
485  int rootId, partId;
486 
487  // Final processing of coefficients of waves in stars.
488  // Require that the root wave of each star and its negation
489  // have conjugate coefficients, and that each basis function
490  // is normalized.
491 
492  for (i = 0; i < nStar_; ++i) {
493 
494  // Treat open and closed stars separately
495 
496  if (stars_[i].invertFlag == 0) {
497 
498  rootId = stars_[i].beginId;
499  stars_[i].waveBz = waves_[rootId].indicesBz;
500 
501  if (stars_[i].cancel) {
502 
503  // If the star is closed set all coefficients to zero
504  std::complex<double> czero(0.0, 0.0);
505  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
506  waves_[j].coeff = czero;
507  }
508 
509  } else {
510 
511  // Compute negation of root vector, shift to DFT mesh
512  nVec.negate(waves_[rootId].indicesBz);
513  (*meshPtr_).shift(nVec);
514 
515  // Find negation of root in this star, set partId to index
516  bool negationFound = false;
517  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
518  if (nVec == waves_[j].indicesDft) {
519  partId = j;
520  negationFound = true;
521  break;
522  }
523  }
524  UTIL_CHECK(negationFound);
525 
526  // Divide all coefficients by the root coefficient
527  rootCoeff = waves_[rootId].coeff;
528  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
529  waves_[j].coeff /= rootCoeff;
530  }
531 
532  if (partId != rootId) {
533 
534  // Compute common divisor
535  partCoeff = waves_[partId].coeff;
536  d = sqrt(partCoeff);
537  if (abs(imag(d)) > 1.0E-10) {
538  if (imag(d) < 0.0) {
539  d = -d;
540  }
541  }
542 
543  // Divide all coefficients by constant divisor
544  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
545  waves_[j].coeff /= d;
546  }
547 
548  }
549 
550  } // end if (cancel) ... else ...
551 
552  } else
553  if (stars_[i].invertFlag == 1) {
554 
555  // Process a pair of open stars related by inversion.
556  // Preconditions:
557  UTIL_CHECK(stars_[i].size == stars_[i+1].size);
558  UTIL_CHECK(stars_[i].cancel == stars_[i+1].cancel);
559 
560  // Identify root of this star (star i)
561  rootId = stars_[i].beginId;
562  stars_[i].waveBz = waves_[rootId].indicesBz;
563 
564  // Compute negation of root vector, shift to DFT mesh
565  nVec.negate(waves_[rootId].indicesBz);
566  (*meshPtr_).shift(nVec);
567 
568  // Find negation of root wave in the next star (star i+1)
569  bool negationFound = false;
570  for (j = stars_[i+1].beginId; j < stars_[i+1].endId; ++j) {
571  if (nVec == waves_[j].indicesDft) {
572  partId = j;
573  stars_[i+1].waveBz = waves_[j].indicesBz;
574  negationFound = true;
575  break;
576  }
577  }
578  UTIL_CHECK(negationFound);
579 
580  if (stars_[i].cancel) {
581 
582  std::complex<double> czero(0.0, 0.0);
583  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
584  waves_[j].coeff = czero;
585  }
586  for (j = stars_[i+1].beginId; j < stars_[i+1].endId; ++j) {
587  waves_[j].coeff = czero;
588  }
589 
590  } else {
591 
592  // Divide all coefficients in this star by root coeff
593  rootCoeff = waves_[rootId].coeff;
594  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
595  waves_[j].coeff /= rootCoeff;
596  }
597 
598  // Divide coefficients in next star by a partner coeff.
599  partCoeff = waves_[partId].coeff;
600  for (j = stars_[i+1].beginId; j < stars_[i+1].endId; ++j) {
601  waves_[j].coeff /= partCoeff;
602  }
603 
604  } // end if (cancel) ... else ...
605 
606  } // end if (invertFlag == -1)
607 
608  } // end loop over stars
609 
610  // Final processing of waves in stars
611  for (i = 0; i < nStar_; ++i) {
612 
613  // Set starId and normalize coefficients for associated waves
614  double snorm = 1.0/sqrt(double(stars_[i].size));
615  for (j = stars_[i].beginId; j < stars_[i].endId; ++j) {
616  waves_[j].starId = i;
617  waves_[j].coeff *= snorm;
618  }
619 
620  }
621 
622  // Final processing of waves
623  for (i = 0; i < nWave_; ++i) {
624  vec = waves_[i].indicesDft;
625 
626  // Validity check
627  for (j = 0; j < D; ++j) {
628  UTIL_CHECK(vec[j] >= 0);
629  UTIL_CHECK(vec[j] < meshDimensions[j]);
630  }
631 
632  // Set implicit attribute
633  if ((vec[D-1] + 1) > (meshDimensions[D-1]/2 + 1)) {
634  waves_[i].implicit = true;
635  } else {
636  waves_[i].implicit = false;
637  }
638 
639  // Look up table for waves
640  waveIds_[mesh().rank(vec)] = i;
641  }
642 
643  }

705  template <int D>
706  void Basis<D>::outputStars(std::ostream& out, bool outputAll) const
707  {
708  // Output number of stars in appropriate format
709  if (outputAll) {
710  out << "N_star" << std::endl
711  << " " << nStar_ << std::endl;
712  } else {
713  out << "N_basis" << std::endl
714  << " " << nBasis_ << std::endl;
715  }
716 
717  // Loop over stars
