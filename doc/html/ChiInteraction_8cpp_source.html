29  * Read chi matrix from file.
30  */
31  void ChiInteraction::readParameters(std::istream& in)
32  {
33  UTIL_CHECK(nMonomer() > 0);
34  chi_.allocate(nMonomer(), nMonomer());
35  chiInverse_.allocate(nMonomer(), nMonomer());
36  idemp_.allocate(nMonomer(), nMonomer());
37  for (int i = 0; i < nMonomer(); ++i)
38  {
39  for (int j = 0; j < nMonomer(); ++j)
40  {
41  chi_(i,j) = 0.0;
42  }
43  }
44  readDSymmMatrix(in, "chi", chi_, nMonomer());
45 
46  for (int i = 0; i < nMonomer(); ++i)
47  {
48  for (int j = 0; j < nMonomer(); ++j)
49  {
50  chi_(i,j) *= nSegment();
51  }
52  }
53 
54  if (nMonomer() == 2) {
55  double det = chi_(0,0)*chi_(1, 1) - chi_(0,1)*chi_(1,0);
56  double norm = chi_(0,0)*chi_(0, 0) + chi_(1,1)*chi_(1,1)
57  + 2.0*chi_(0,1)*chi_(1,0);
58  if (fabs(det/norm) < 1.0E-10) {
59  UTIL_THROW("Singular chi matrix");
60  }
61  chiInverse_(0,1) = -chi_(0,1)/det;
62  chiInverse_(1,0) = -chi_(1,0)/det;
63  chiInverse_(1,1) = chi_(0,0)/det;
64  chiInverse_(0,0) = chi_(1,1)/det;
65 
66  } else {
67  LuSolver solver;
68  solver.allocate(nMonomer());
69  solver.computeLU(chi_);
70  solver.inverse(chiInverse_);
71  }
72 
73  double sum = 0;
74  int i, j, k;
75 
76  for (i = 0; i < nMonomer(); ++i) {
77  idemp_(0,i) = 0;
78  for (j = 0; j < nMonomer(); ++j) {
79  idemp_(0,i) -= chiInverse_(j,i);
80  }
81  sum -= idemp_(0,i);
82  for (k = 0; k < nMonomer(); ++k) { //row
83  idemp_(k,i) = idemp_(0,i);
84  }
85  }
86 
87  for (i = 0; i < nMonomer(); ++i) { //row
88  for (j = 0; j < nMonomer(); ++j) { //coloumn
89  idemp_(i,j) /= sum;
90  }
91  idemp_(i,i) +=1.0;
92  }
93 
94  sum_inv_ = sum;
95 
96  }

100  */
101  double ChiInteraction::fHelmholtz(Array<double> const & c) const
102  {
103  int i, j;
104  double sum = 0.0;
105  for (i = 0; i < nMonomer(); ++i) {
106  for (j = 0; j < nMonomer(); ++j) {
107  sum += chi_(i, j)* c[i]*c[j];
108  }
109  }
110  return 0.5*sum;

130  * Compute concentrations and xi from chemical potentials.
131  */
132  void
133  ChiInteraction::computeC(Array<double> const & w,
134  Array<double>& c, double& xi)
135  const
136  {
137  double sum1 = 0.0;
138  double sum2 = 0.0;
139  int i, j;
140  for (i = 0; i < nMonomer(); ++i) {
141  for (j = 0; j < nMonomer(); ++j) {
142  sum1 += chiInverse_(i, j)*w[j];
143  sum2 += chiInverse_(i, j);
144  }
145  }
146  xi = (sum1 - 1.0)/sum2;
147  for (i = 0; i < nMonomer(); ++i) {
148  c[i] = 0;
149  for (j = 0; j < nMonomer(); ++j) {
150  c[i] += chiInverse_(i, j)*( w[j] - xi );
151  }
152  }
