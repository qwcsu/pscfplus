218  // Polymers
219  read<int>(in, "nPolymer", nPolymer_);
220  polymers_.allocate(nPolymer_);
221  for (int i = 0; i < nPolymer_; ++i)
222  {
223  readParamComposite(in, polymers_[i]);
224  }
225 
226  // Set statistical segment lengths for all bonds
227  double kuhn1, kuhn2;
228  int monomerId1, monomerId2;
229  for (int i = 0; i < nPolymer_; ++i)
230  {
231  for (int j = 0; j < polymer(i).nBond(); ++j)
232  {
233  monomerId1 = polymer(i).bond(j).monomerId(0);
234  monomerId2 = polymer(i).bond(j).monomerId(1);
235  if (monomerId1 >= 0 && monomerId2 >= 0)
236  {
237  kuhn1 = monomer(monomerId1).step();
238  kuhn2 = monomer(monomerId2).step();
239  polymer(i).bond(j).setKuhn(kuhn1, kuhn2);
240  }
241  else if (monomerId1 >= 0 && monomerId2 < 0)
242  {
243  kuhn1 = monomer(monomerId1).step();
244  polymer(i).bond(j).setKuhn(kuhn1, kuhn1);
245  }
246  else if (monomerId1 < 0 && monomerId2 >= 0)
247  {
248  kuhn2 = monomer(monomerId2).step();
249  polymer(i).bond(j).setKuhn(kuhn2, kuhn2);
