63  for (int i = 0; i < D; ++i) {
64  UTIL_CHECK(rDimensions[i] > 0);
65  meshDimensions_[i] = rDimensions[i];
66  rSize_ *= rDimensions[i];
67  if (i < D - 1) {
68  kSize_ *= rDimensions[i];
69  } else {
70  kSize_ *= (rDimensions[i]/2 + 1);
71  }
72  }

75  UTIL_CHECK(kField.capacity() == kSize_);
76
