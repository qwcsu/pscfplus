89  * Set molecular and monomer volume fractions.

91  void Mixture::setComposition(DArray<double> const & phi)
92  {
93  validate();
94  UTIL_ASSERT(phi.capacity() == nMolecule_);
95 
96  // Set molecular volume fractions
97  double sum = 0;
98  for (int i = 0; i < nMolecule_ - 1; ++i) {
99  UTIL_CHECK(phi[i] >= 0.0);
100  UTIL_CHECK(phi[i] <= 1.0);
101  phi_[i] = phi[i];
102  sum += phi[i];
103  }
104  UTIL_CHECK(sum <= 1.0);
105  phi_[nMolecule_ -1] = 1.0 - sum;
106 
107  computeC();
108  hasComposition_ = true;
109  }

111  void Mixture::computeC()
112  {
113  // Initialize monomer fractions to zero
114  int k;
115  for (k = 0; k < nMonomer_; ++k) {
116  c_[k] = 0.0;
117  }
118 
119  // Compute monomer volume fractions
120  double concentration;
121  int i, j;
122  for (i = 0; i < nMolecule_; ++i) {
123  Molecule& mol = molecules_[i];
124  concentration = phi_[i]/mol.size();
125  for (j = 0; j < molecules_[i].nClump(); ++j) {
126  k = mol.clump(j).monomerId();
127  c_[k] += concentration*mol.clump(j).size();
128  }
129  }
130 
131  // Check and normalize monomer volume fractions
132  double sum = 0.0;
133  for (k = 0; k < nMonomer_; ++k) {
134  UTIL_CHECK(c_[k] >= 0.0);
135  UTIL_CHECK(c_[k] <= 1.0);
136  sum += c_[k];
137  }
138  UTIL_CHECK(sum > 0.9999);
139  UTIL_CHECK(sum < 1.0001);
140  for (k = 0; k < nMonomer_; ++k) {
141  c_[k] /= sum;
142  }
143 
144  }

146  /*
147  * Compute thermodynamic properties.
148  */
149  void Mixture::computeMu(Interaction const & interaction,
150  double xi)
151  {
152  UTIL_CHECK(interaction.nMonomer() == nMonomer_);
153  UTIL_CHECK(hasComposition_);
154 
155  // Compute monomer excess chemical potentials
156  interaction.computeW(c_, w_);
157 
158  int m; // molecule index
159  int c; // clump index
160  int t; // monomer type index
161  double mu, size;
162  for (m = 0; m < nMolecule_; ++m) {
163  Molecule& mol = molecules_[m];
164  mu = log( phi_[m] );
165  mu += xi*mol.size();
166  for (c = 0; c < mol.nClump(); ++c) {
167  t = mol.clump(c).monomerId();
168  size = mol.clump(c).size();
169  mu += size*w_[t];
170  }
171  mu_[m] = mu;
172 
173  }

178  * Compute composition from chemical potentials.
179  */
180  void Mixture::computePhi(Interaction const & interaction,
181  DArray<double> const & mu,
182  DArray<double> const & phi,
183  double& xi)
184  {
185  UTIL_ASSERT(interaction.nMonomer() == nMonomer_);
186 
187  // Allocate residual and jacobian on first use.
188  if (residual_.capacity() == 0) {
189  residual_.allocate(nMonomer_);
190  dX_.allocate(nMolecule_);
191  dWdC_.allocate(nMonomer_, nMonomer_);
192  dWdPhi_.allocate(nMonomer_, nMolecule_);
193  jacobian_.allocate(nMolecule_, nMolecule_);
194  phiOld_.allocate(nMolecule_);
195  solverPtr_ = new LuSolver();
196  solverPtr_->allocate(nMolecule_);
197  }
198 
199  // Compute initial state
200  setComposition(phi);
201  computeMu(interaction, xi);
202  adjustXi(mu, xi);
203 
204  // Compute initial residual
205  double error;
206  double epsilon = 1.0E-10;
207  computeResidual(mu, error);
208 
209  #if 0
210  std::cout << "mu[0] =" << mu[0] << std::endl;
211  std::cout << "mu[1] =" << mu[1] << std::endl;
212  std::cout << "mu_[0] =" << mu_[0] << std::endl;
213  std::cout << "mu_[1] =" << mu_[1] << std::endl;
214  std::cout << "residual[0] =" << residual_[0] << std::endl;
215  std::cout << "residual[1] =" << residual_[1] << std::endl;
216  std::cout << "error =" << error << std::endl;
217  #endif
218 
219  if (error < epsilon) return;
220 
221  double s1; // clump size
222  double v1; // molecule size
223  double f1; // clump fraction
224  int m1, m2; // molecule type indices
225  int c1; // clump index
226  int t1, t2; // monomer type indices
227 
228  for (int it = 0; it < 50; ++it) {
229 
230  // Compute matrix of derivative dWdC (C = monomer fraction)
231  interaction.computeDwDc(c_, dWdC_);
232 
233  // Compute matrix derivative dWdPhi (Phi = molecule fraction)
234  for (t1 = 0; t1 < nMonomer_; ++t1) {
235  for (m1 = 0; m1 < nMolecule_; ++m1) {
236  dWdPhi_(t1, m1) = 0.0;
237  }
238  }
239  for (m1 = 0; m1 < nMolecule_; ++m1) {
240  v1 = molecule(m1).size();
241  for (c1 = 0; c1 < molecule(m1).nClump(); ++c1) {
242  t1 = molecule(m1).clump(c1).monomerId();
243  s1 = molecule(m1).clump(c1).size();
244  f1 = s1/v1;
245  for (t2 = 0; t2 < nMonomer_; ++t2) {
246  dWdPhi_(t2, m1) += dWdC_(t2, t1)*f1;
247  }
248  }
249  }
250 
251  // Compute matrix d(mu)/d(Phi), stored in jacobian_
252  for (m1 = 0; m1 < nMolecule_; ++m1) {
253  for (m2 = 0; m2 < nMolecule_; ++m2) {
254  jacobian_(m1, m2) = 0.0;
255  }
256  jacobian_(m1, m1) = 1.0/phi_[m1];
257  }
258  for (m1 = 0; m1 < nMolecule_; ++m1) {
259  v1 = molecule(m1).size();
260  for (c1 = 0; c1 < molecule(m1).nClump(); ++c1) {
261  t1 = molecule(m1).clump(c1).monomerId();
262  s1 = molecule(m1).clump(c1).size();
263  for (m2 = 0; m2 < nMolecule_; ++m2) {
264  jacobian_(m1, m2) += s1*dWdPhi_(t1, m2);
265  }
266  }
267  }
268 
269  // Impose incompressibility
270  int mLast = nMolecule_ - 1;
271  for (m1 = 0; m1 < nMolecule_; ++m1) {
272  for (m2 = 0; m2 < nMolecule_ - 1; ++m2) {
273  jacobian_(m1, m2) -= jacobian_(m1, mLast);
274  }
275  // Derivative of mu_[m1] with respect to xi
276  jacobian_(m1, mLast) = molecule(m1).size();
277  }
278 
279  // Newton Raphson update of phi and xi fields
280  solverPtr_->computeLU(jacobian_);
281  solverPtr_->solve(residual_, dX_);
282 
283  // Store old value of phi
284  for (m1 = 0; m1 < nMolecule_; ++m1) {
285  phiOld_[m1] = phi_[m1];
286  }
287 
288  // Apply update
289  bool inRange = false;
290  for (int j = 0; j < 5; ++j) {
291 
292  // Update volume fractions
293  double sum = 0.0;
294  for (m1 = 0; m1 < nMolecule_ - 1; ++m1) {
295  phi_[m1] = phiOld_[m1] - dX_[m1];
296  sum += phi_[m1];
297  }
298  phi_[mLast] = 1.0 - sum;
299 
300  // Check if all volume fractions are in [0,1]
301  inRange = true;
302  for (m1 = 0; m1 < nMolecule_; ++m1) {
303  if (phi_[m1] < 0.0) inRange = false;
304  if (phi_[m1] > 1.0) inRange = false;
305  }
306 
307  // Exit loop or reduce increment
308  if (inRange) {
309  break;
310  } else {
311  for (m1 = 0; m1 < nMolecule_; ++m1) {
312  dX_[m1] *= 0.5;
313  }
314  }
315 
316  }
317  if (inRange) {
318  xi = xi - dX_[mLast];
319  } else {
320  UTIL_THROW("Volume fractions remain out of range");
321  }
322 
323  // Compute residual
324  computeC();
325  computeMu(interaction, xi);
326  adjustXi(mu, xi);
327  computeResidual(mu, error);
328 
329  #if 0
330  std::cout << "mu[0] =" << mu[0] << std::endl;
331  std::cout << "mu[1] =" << mu[1] << std::endl;
332  std::cout << "mu_[0] =" << mu_[0] << std::endl;
333  std::cout << "mu_[1] =" << mu_[1] << std::endl;
334  std::cout << "residual[0] =" << residual_[0] << std::endl;
335  std::cout << "residual[1] =" << residual_[1] << std::endl;
336  std::cout << "error =" << error << std::endl;
337  #endif
338 
339  if (error < epsilon) return;
340  }
341 
342  UTIL_THROW("Failed to converge");
343  }

345  void Mixture::adjustXi(DArray<double> const & mu, double& xi)
346  {
347  double dxi = 0.0;
348  double sum = 0.0;
349  for (int i=0; i < nMolecule_; ++i) {
350  dxi += mu[i] - mu_[i];
351  sum += molecules_[i].size();
352  }
353  dxi = dxi/sum;
354  for (int i=0; i < nMolecule_; ++i) {
355  mu_[i] += dxi*molecules_[i].size();
356  }
357  xi += dxi;
358  }

371  /*
372  * Compute Helmoltz free energy and pressure
373  */
374  void Mixture::computeFreeEnergy(Interaction const & interaction)
375  {
376  fHelmholtz_ = 0.0;
377 
378  // Compute ideal gas contributions to fHelhmoltz_
379  double size;
380  for (int i = 0; i < nMolecule_; ++i) {
381  size = molecules_[i].size();
382  fHelmholtz_ += phi_[i]*( log(phi_[i]) - 1.0 )/size;
383  }
384 
385  // Add average interaction free energy density per monomer
386  fHelmholtz_ += interaction.fHelmholtz(c_);
387 
388  // Compute pressure
389  pressure_ = -fHelmholtz_;
390  for (int i = 0; i < nMolecule_; ++i) {
391  size = molecules_[i].size();
392  pressure_ += phi_[i]*mu_[i]/size;
393  }
394 
395  }
