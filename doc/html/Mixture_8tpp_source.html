489

491  for (int p = 0; p < np; ++p)
492  {
493  int nb = polymer(p).nBlock();

502  (tmp, polymer(p).block(b).cField().cDField(), rSize_);
503  sBlock_[idx] = -(polymer(p).phi()/polymer(p).length()/mesh.size())*gpuSum(tmp, rSize_);
504

564  int nb2 = polymer(p2).nBlock();
565  for (int b1 = 0; b1 < nb1; ++b1)

568  {
569  if (p1 == p2 && b1 == b2)

571  break;
572  }

579  chi = interaction_->chi(m1, m2);

581  if (chi != 0.0)
582  {
583  list[0] = p1;
584  list[1] = b1;

586  list[3] = b2;
