344  blocks_.allocate(nBlock_);
345  vertices_.allocate(nVertex_);
346  propagatorIds_.allocate(2 * nBlock_);
347 
348  readDArray<Block>(in, "blocks", blocks_, nBlock_);
349 
350  // Set vertex indices
351  for (int vertexId = 0; vertexId < nVertex_; ++vertexId)
352  {
353  vertices_[vertexId].setId(vertexId);
354  }
355 
356  // Add blocks to vertices
357  int vertexId0, vertexId1;
358  Block *blockPtr;
359  for (int blockId = 0; blockId < nBlock_; ++blockId)
360  {

362  vertexId0 = blockPtr->vertexId(0);
363  vertexId1 = blockPtr->vertexId(1);
364  vertices_[vertexId0].addBlock(*blockPtr);
365  vertices_[vertexId1].addBlock(*blockPtr);

368  makePlan();

371  ensemble_ = Species::Closed;

373  if (ensemble_ == Species::Closed)

379  read(in, "mu", mu_);

417  template <class Block>
418  void PolymerTmpl<Block>::makePlan()
419  {
420  if (nPropagator_ != 0)
421  {
422  UTIL_THROW("nPropagator !=0 on entry");
423  }
424 
425  // Allocate and initialize isFinished matrix
426  DMatrix<bool> isFinished;
427  isFinished.allocate(nBlock_, 2);
428  for (int iBlock = 0; iBlock < nBlock_; ++iBlock)
429  {
430  for (int iDirection = 0; iDirection < 2; ++iDirection)
431  {
432  isFinished(iBlock, iDirection) = false;
433  }
434  }
435 
436  Pair<int> propagatorId;
437  Vertex *inVertexPtr = nullptr;
438  int inVertexId = -1;
439  bool isReady;
440  while (nPropagator_ < nBlock_ * 2)
441  {
442  for (int iBlock = 0; iBlock < nBlock_; ++iBlock)
443  {
444  for (int iDirection = 0; iDirection < 2; ++iDirection)
445  {
446  if (!isFinished(iBlock, iDirection))
447  {
448  inVertexId = blocks_[iBlock].vertexId(iDirection);
449  inVertexPtr = &vertices_[inVertexId];
450  isReady = true;
451  for (int j = 0; j < inVertexPtr->size(); ++j)
452  {
453  propagatorId = inVertexPtr->inPropagatorId(j);
454  if (propagatorId[0] != iBlock)
455  {
456  if (!isFinished(propagatorId[0], propagatorId[1]))
457  {
458  isReady = false;
459  break;
460  }
461  }
462  }
463  if (isReady)
464  {
465  propagatorIds_[nPropagator_][0] = iBlock;
466  propagatorIds_[nPropagator_][1] = iDirection;
467  isFinished(iBlock, iDirection) = true;
468  ++nPropagator_;
469  }
470  }
471  }
472  }
473  }

479  template <class Block>
480  void PolymerTmpl<Block>::solve()
481  {
482  int nx = blocks_[0].mesh().size();
483  int NUMBER_OF_BLOCKS, THREADS_PER_BLOCK;
484  Pspg::ThreadGrid::setThreadsLogical(nx, NUMBER_OF_BLOCKS, THREADS_PER_BLOCK);
485  // Clear all propagators
486  for (int j = 0; j < nPropagator(); ++j)
487  {
488  propagator(j).setIsSolved(false);
489  }
490 
491  DArray<Pair<int>> t;
492  int partner[nBlock()];
493  t.allocate(nBlock());
494  for (int i = 0; i < nBlock(); ++i)
495  {
496  for (int j = 0; j < nBlock(); ++j)
497  {
498  if (i == propagatorIds_[j][0])
499  {
500  t[i][0] = j;
501  // std::cout << t[i][0] << "\n";
502  }
503  }
504  }
505  // std::cout << "\n";
506  for (int i = 0; i < nBlock(); ++i)
507  {
508  for (int j = nBlock(); j < nPropagator(); ++j)
509  {
510  if (i == propagatorIds_[j][0])
511  {
512  t[i][1] = j;
513  // std::cout << t[i][0] << "\n";
514  }
515  }
516  }
517  for (int j = 0; j < nBlock(); ++j)
518  {
519  partner[t[j][1] - nBlock()] = t[j][0];
520  // std::cout << t[j][1]-nBlock() << " " << t[j][0] << "\n";
521  }
522  t.deallocate();
523 
524  for (int j = 0; j < nBlock(); ++j)
525  {
526  UTIL_CHECK(propagator(j).isReady())
527  propagator(j).solveForward();
528  }
529 
530  for (int j = nBlock(); j < nPropagator(); ++j)
531  {
532  UTIL_CHECK(propagator(j).isReady())
533  propagator(j).solveBackward(propagator(partner[j - nBlock()]).qhead(), j - nBlock());
534  }
535 
536  Q_ = propagator(nBlock()).returnQ();
537 
538  double scale = 1.0 / Q_;
539 
540  for (int j = 0; j < nBlock(); ++j)
541  {
542  Pspg::scaleReal<<<NUMBER_OF_BLOCKS, THREADS_PER_BLOCK>>>(blocks_[j].cField().cDField(), scale, nx);
543  }
544 
545  if (ensemble() == Species::Closed)
546  {
547  mu_ = log(phi_ / Q_);
548  }
549  else if (ensemble() == Species::Open)
550  {
551  phi_ = exp(mu_) * Q_;
552  }
553  }
554 
555 
556 }
