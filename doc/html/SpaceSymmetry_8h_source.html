378  operator * (const SpaceSymmetry<D>& A, const SpaceSymmetry<D>& B)
379  {
380  SpaceSymmetry<D> C;
381  int i, j, k;
382 
383  // Compute rotation matrix (matrix product)
384  for (i = 0; i < D; ++i) {
385  for (j = 0; j < D; ++j) {
386  C.R_(i, j) = 0;
387  for (k = 0; k < D; ++k) {
388  C.R_(i, j) += A.R_(i, k)*B.R_(k, j);
389  }
390  }
391  }
392 
393  // Compute translation vector
394  for (i = 0; i < D; ++i) {
395  C.t_[i] = A.t_[i];
396  }
397  for (i = 0; i < D; ++i) {
398  for (j = 0; j < D; ++j) {
399  C.t_[i] += A.R_(i, j)*B.t_[j];
400  }
401  }
402  C.normalize();
403
