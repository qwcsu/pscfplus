54  /*
55  * Set the Bravais and reciprocal lattice parameters.
56  */
57  void UnitCell<3>::setBasis()
58  {
59  UTIL_CHECK(lattice_ != UnitCell<3>::Null);
60  UTIL_CHECK(nParameter_ > 0);
61 
62  // Set elements for specific lattice types
63  double twoPi = 2.0*Constants::Pi;
64  int i;
65  if (lattice_ == UnitCell<3>::Cubic) {
66  UTIL_CHECK(nParameter_ == 1);
67  for (i=0; i < 3; ++i) {
68  rBasis_[i][i] = parameters_[0];
69  kBasis_[i][i] = twoPi/parameters_[0];
70  drBasis_[0](i,i) = 1.0;
71  }
72  } else
73  if (lattice_ == UnitCell<3>::Tetragonal) {
74  UTIL_CHECK(nParameter_ == 2);
75  rBasis_[0][0] = parameters_[0];
76  rBasis_[1][1] = parameters_[0];
77  rBasis_[2][2] = parameters_[1];
78 
79  kBasis_[0][0] = twoPi/parameters_[0];
80  kBasis_[1][1] = kBasis_[0][0];
81  kBasis_[2][2] = twoPi/parameters_[1];
82 
83  drBasis_[0](0,0) = 1.0;
84  drBasis_[0](1,1) = 1.0;
85  drBasis_[1](2,2) = 1.0;
86  } else
87  if (lattice_ == UnitCell<3>::Orthorhombic) {
88  UTIL_CHECK(nParameter_ == 3);
89  for (i=0; i < 3; ++i) {
90  rBasis_[i][i] = parameters_[i];
91  drBasis_[i](i,i) = 1.0;
92  kBasis_[i][i] = twoPi/parameters_[i];
93  }
94  } else
95  if (lattice_ == UnitCell<3>::Hexagonal) {
96  UTIL_CHECK(nParameter_ == 2);
97  double a = parameters_[0];
98  double c = parameters_[1];
99  double rt3 = sqrt(3.0);
100 
101  rBasis_[0][0] = a;
102  rBasis_[0][1] = 0.0;
103  rBasis_[1][0] = -0.5*a;
104  rBasis_[1][1] = 0.5*rt3*a;
105  rBasis_[2][2] = c;
106 
107  drBasis_[0](0, 0) = 1.0;
108  drBasis_[0](0, 1) = 0.0;
109  drBasis_[0](1, 0) = -0.5;
110  drBasis_[0](1, 1) = 0.5*rt3;
111  drBasis_[1](2, 2) = 1.0;
112 
113  kBasis_[0][0] = twoPi/a;
114  kBasis_[0][1] = twoPi/(rt3*a);
115  kBasis_[1][0] = 0.0;
116  kBasis_[1][1] = twoPi/(0.5*rt3*a);
117  kBasis_[2][2] = twoPi/(c);
118 
119  }else
120  if (lattice_ == UnitCell<3>::Rhombohedral) {
121  UTIL_CHECK(nParameter_ == 2);
122 
123  // Set parameters
124  double a, beta, theta;
125  a = parameters_[0]; // magnitude of Bravais basis vectors
126  beta = parameters_[1]; // angle between basis vectors
127  theta = acos( sqrt( (2.0*cos(beta) + 1.0)/3.0 ) );
128  // theta is the angle of all basis vectors from the z axis
129 
130  /*
131  * Bravais lattice vectora a_i with i=0, 1, or 2 have form:
132  *
133  * a_i = a * (z * cos(theta) + u_i * sin(theta) )
134  *
135  * where z denotes a unit vector parallel to the z axis, and
136  * the u_i are three unit vectors in the x-y separated by
137  * angles of 2 pi /3 (or 120 degrees), for which u_0 is
138  * parallel to the x axis. Note that u_i . u_j = -1/2 for any
139  * i not equal to j.
140  *
141  * The angle between any two such Bravais basis vectors is easily
142  * computed from the dot product, a_i . a_j for unequal i and j.
143  * This gives:
144  *
145  * cos beta = cos^2(theta) - 0.5 sin^2 (theta)
146  * = (3cos^{2}(theta) - 1)/2
147  *
148  * The angle theta is computed above by solving this equation
149  * for theta as a function of the input parameter beta.
150  *
151  * The corresponding reciprocal lattice vectors have the form
152  *
153  * b_i = (2*pi/3*a)( z/cos(theta) + 2 * u_i / sin(theta) )
154  *
155  * It is straightforward to confirm that these reciprocal
156  * vectors satisfy the condition a_i . b_j = 2 pi delta_{ij}
157  */
158 
159  // Trig function values
160  double cosT = cos(theta);
161  double sinT = sin(theta);
162  double sqrt3 = sqrt(3.0);
163  double sinPi3 = 0.5*sqrt3;
164  double cosPi3 = 0.5;
165 
166  // Bravais basis vectors
167  rBasis_[0][0] = sinT * a;
168  rBasis_[0][2] = cosT * a;
169  rBasis_[1][0] = -cosPi3 * sinT * a;
170  rBasis_[1][1] = sinPi3 * sinT * a;
171  rBasis_[1][2] = cosT * a;
172  rBasis_[2][0] = -cosPi3 * sinT * a;
173  rBasis_[2][1] = -sinPi3 * sinT * a;
174  rBasis_[2][2] = cosT * a;
175 
176  // Reciprocal lattice basis vectors
177  kBasis_[0][0] = 2.0 * twoPi /( 3.0 * sinT * a);
178  kBasis_[0][2] = twoPi /( 3.0 * cosT * a);
179  kBasis_[1][0] = -2.0 * cosPi3 * twoPi /( 3.0 * sinT * a);
180  kBasis_[1][1] = 2.0 * sinPi3 * twoPi /( 3.0 * sinT * a);
181  kBasis_[1][2] = twoPi /( 3.0 * cosT * a);
182  kBasis_[2][0] = -2.0 * cosPi3 * twoPi /( 3.0 * sinT * a);
183  kBasis_[2][1] = -2.0 * sinPi3 * twoPi /( 3.0 * sinT * a);
184  kBasis_[2][2] = twoPi /( 3.0 * cosT * a);
185 
186  // Derivatives with respect to length a
187  drBasis_[0](0,0) = sinT;
188  drBasis_[0](0,2) = cosT;
189  drBasis_[0](1,0) = -cosPi3 * sinT;
190  drBasis_[0](1,1) = sinPi3 * sinT;
191  drBasis_[0](1,2) = cosT;
192  drBasis_[0](2,0) = -cosPi3 * sinT;
193  drBasis_[0](2,1) = -sinPi3* sinT;
194  drBasis_[0](2,2) = cosT;
195 
196  // Define alpha = d(theta)/d(beta)
197  double alpha = -sin(beta)/(3.0*cosT*sinT);
198 
199  // Derivatives with respect to beta
200  drBasis_[1](0,0) = cosT * alpha * a;
201  drBasis_[1](0,2) = -sinT * alpha * a;
202  drBasis_[1](1,0) = -cosPi3 * cosT * alpha * a;
203  drBasis_[1](1,1) = sinPi3 * cosT * alpha * a;
204  drBasis_[1](1,2) = -sinT * alpha * a;
205  drBasis_[1](2,0) = -cosPi3 * cosT * alpha * a;
206  drBasis_[1](2,1) = -sinPi3* cosT * alpha * a;
207  drBasis_[1](2,2) = -sinT * alpha * a;
208 
209  } else
210  if (lattice_ == UnitCell<3>::Orthorhombic) {
211  UTIL_CHECK(nParameter_ == 3);
212  for (i=0; i < 3; ++i) {
213  rBasis_[i][i] = parameters_[i];
214  kBasis_[i][i] = twoPi/parameters_[i];
215  drBasis_[i](i,i) = 1.0;
216  }
217  } else
218  if (lattice_ == UnitCell<3>::Monoclinic) {
219  UTIL_CHECK(nParameter_ == 4);
220 
221  /*
222  * Description: Bravais basis vectors A, B, and C
223  * A (vector 0) is along x axis
224  * B (vector 1) is along y axis
225  * C (vector 2) is in the x-z plane, an angle beta from A/x
226  * B is the unique axis (perpendicular to A and C)
227  * Angle beta is stored in radians.
228  */
229 
230  // Parameters
231  double a = parameters_[0]; // length of A
232  double b = parameters_[1]; // length of B
233  double c = parameters_[2]; // length of C
234  double beta = parameters_[3]; // angle between C and A/x
235 
236  double cb = cos(beta);
237  double sb = sin(beta);
238 
239  // Nonzero components of basis vectors
240  // For rBasis_[i][j], i:basis vector, j:Cartesian component
241  rBasis_[0][0] = a;
242  rBasis_[1][1] = b;
243  rBasis_[2][0] = c*cb;
244  rBasis_[2][2] = c*sb;
245 
246  // Nonzero derivatives of basis vectors with respect to parameters
247  // For drBasis_[k](i,j), k:parameter, i:basis vector, j:Cartesian
248  drBasis_[0](0,0) = 1.0;
249  drBasis_[1](1,1) = 1.0;
250  drBasis_[2](2,0) = cb;
251  drBasis_[2](2,2) = sb;
252  drBasis_[3](2,0) = c*sb;
253  drBasis_[3](2,2) = -c*cb;
254 
255  // Reciprocal lattice vectors
256  kBasis_[0][0] = twoPi/a;
257  kBasis_[0][2] = -twoPi*cb/(a*sb);
258  kBasis_[1][1] = twoPi / b;
259  kBasis_[2][2] = twoPi/(c*sb);
260 
261  } else
262  if (lattice_ == UnitCell<3>::Triclinic) {
263  UTIL_CHECK(nParameter_ == 6);
264 
265  /*
266  * Description: Bravais basis vectors A, B, and C
267  * A (vector 0) is along x axis
268  * B (vector 1) is in the x-y plane, an angle gamma from x axis
269  * C (vector 2) is tilted by an angle theta from z axis
270  * phi is the angle beween c-z an x-z (or a-z) planes
271  * All three angles are stored in radians
272  */
273  double a = parameters_[0]; // length of A
274  double b = parameters_[1]; // length of B
275  double c = parameters_[2]; // length of C
276  double phi = parameters_[3]; // angle between c-z and a-z planes
277  double theta = parameters_[4]; // angle between c and z axis
278  double gamma = parameters_[5]; // angle between a and b
279 
280  // sine and cosine of all angles
281  double cosPhi = cos(phi);
282  double sinPhi = sin(phi);
283  double cosTheta = cos(theta);
284  double sinTheta = sin(theta);
285  double cosGamma = cos(gamma);
286  double sinGamma = sin(gamma);
287 
288  // Nonzero components of Bravais basis vectors
289  rBasis_[0][0] = a;
290  rBasis_[1][0] = b * cosGamma;
291  rBasis_[1][1] = b * sinGamma;
292  rBasis_[2][0] = c * sinTheta * cosPhi;
293  rBasis_[2][1] = c * sinTheta * sinPhi;
294  rBasis_[2][2] = c * cosTheta;
295 
296  // Nonzero derivatives of basis vectors with respect to parameters
297  drBasis_[0](0,0) = 1.0;
298  drBasis_[1](1,0) = cosGamma;
299  drBasis_[1](1,1) = sinGamma;
300  drBasis_[2](2,0) = sinTheta * cosPhi;
301  drBasis_[2](2,1) = sinTheta * sinPhi;
302  drBasis_[2](2,2) = cosTheta;
303  drBasis_[3](2,0) = -c * sinTheta * sinPhi;
304  drBasis_[3](2,1) = c * sinTheta * cosPhi;
305  drBasis_[4](2,0) = c * cosTheta * cosPhi;
306  drBasis_[4](2,1) = c * cosTheta * sinPhi;
307  drBasis_[4](2,2) = -c * sinTheta;
308  drBasis_[5](1,0) = -b * sinGamma;
309  drBasis_[5](1,1) = b * cosGamma;
310 
311  // Reciprocal lattice vectors
312  kBasis_[0][0] = twoPi / a;
313  kBasis_[0][1] = -twoPi * cosGamma / (a * sinGamma);
314  kBasis_[0][2] = sinTheta * (cosGamma * sinPhi - sinGamma * cosPhi);
315  kBasis_[0][2] = twoPi * kBasis_[0][2] / (a * sinGamma * cosTheta);
316 
317  kBasis_[1][1] = twoPi/(b*sinGamma);
318  kBasis_[1][2] = -twoPi*sinPhi*sinTheta/(b*cosTheta*sinGamma);
319 
320  kBasis_[2][2] = twoPi/(c*cosTheta);
321 
322  } else{
323  UTIL_THROW("Unimplemented 3D lattice type");
324  }
