334  {

336  RealVec<D> p;
337  for (int i = 0; i < D; ++i) {
338  p.multiply(kBasis_[i], k[i]);
339  g += p;

397  */
398  template <int D>
399  void UnitCellBase<D>::computeDerivatives()
400  {
401  // Compute dkBasis
402  int p, q, r, s, t;
403  for (p = 0; p < nParameter_; ++p) {
404  for (q = 0; q < D; ++q) {
405  for (r = 0; r < D; ++r) {
406 
407  // Loop over free indices s, t
408  for (s = 0; s < D; ++s) {
409  for (t = 0; t < D; ++t) {
410  dkBasis_[p](q,r)
411  -= kBasis_[q][s]*drBasis_[p](t,s)*kBasis_[t][r];
412  }
413  }
414  dkBasis_[p](q,r) /= 2.0*Constants::Pi;
415 
416  }
417  }
418  }
419 
420  // Compute drrBasis and dkkBasis
421  for (p = 0; p < nParameter_; ++p) {
422  for (q = 0; q < D; ++q) {
423  for (r = 0; r < D; ++r) {
424  for (s = 0; s < D; ++s) {
425  drrBasis_[p](q,r) += rBasis_[q][s]*drBasis_[p](r,s);
426  drrBasis_[p](q,r) += rBasis_[r][s]*drBasis_[p](q,s);
427  dkkBasis_[p](q,r) += kBasis_[q][s]*dkBasis_[p](r,s);
428  dkkBasis_[p](q,r) += kBasis_[r][s]*dkBasis_[p](q,s);
429 
430  }
431  }
